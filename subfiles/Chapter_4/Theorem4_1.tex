\subsection{Theorem 4.1}
\begin{wts}
Suppose that $A$ is a subset of $X$, let $\acc{A}$ be the set of accumulation points of $A$, then
\begin{equation}
    \overline{A}=A\cup\acc{(A)}
\end{equation}
and $A$ is closed if and only if $\acc{(A)}\subseteq A$.
\end{wts}
